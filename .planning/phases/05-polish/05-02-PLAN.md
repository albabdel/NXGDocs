<objective>
Replace the broken Algolia integration with Pagefind — a build-time static search engine with zero external service dependencies.

Purpose: The existing Algolia setup has a split-brain (two conflicting App IDs) and requires a separate crawler service that must be manually triggered. Pagefind generates a search index automatically every time the site builds, is fully self-contained, and needs no API keys, no crawler, and no re-index step after content changes.
Output: Pagefind installed and working, Algolia config removed, search index generated at every build.
</objective>

```ts
import { test, expect } from '@playwright/test';

test.describe('PLSH-01: Pagefind search', () => {
  test('Pagefind search input is present on docs page', async ({ page }) => {
    // Pagefind search is embedded in the docs layout
    await page.goto('/docs');
    // docusaurus-pagefind renders a search input
    const searchInput = page.locator('input.pagefind-ui__search-input, [class*="pagefind-ui__search"]').first();
    // Also check for the Docusaurus search button if Pagefind uses a toggle
    const searchToggle = page.locator('[aria-label*="Search"], button[class*="pagefind"], .navbar__search').first();
    // At least one search element must be visible
    const inputVisible = await searchInput.isVisible().catch(() => false);
    const toggleVisible = await searchToggle.isVisible().catch(() => false);
    expect(inputVisible || toggleVisible).toBeTruthy();
  });

  test('Pagefind search index is served', async ({ page }) => {
    // Pagefind index must be accessible as a static file
    const response = await page.goto('/pagefind/pagefind.js');
    expect(response?.status()).toBe(200);
  });

  test.fixme('search returns results for "getting started"', async ({ page }) => {
    // Enable after Phase 4 content is fully migrated and site is deployed with new Pagefind index
    await page.goto('/docs');
    const searchInput = page.locator('input.pagefind-ui__search-input').first();
    await expect(searchInput).toBeVisible();
    await searchInput.fill('getting started');
    await page.waitForTimeout(500); // Pagefind is client-side, slight debounce
    const results = page.locator('.pagefind-ui__results, .pagefind-ui__result').first();
    await expect(results).toBeVisible();
  });
});
```

Note: The `fixme` test is kept for Phase 4 completion. The two non-fixme tests verify the infrastructure is present and working — these should pass immediately after deployment.
