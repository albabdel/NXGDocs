```typescript
/**
 * Product Configuration
 * 
 * Centralized configuration for multi-product analytics.
 * Add new products by extending the PRODUCTS registry.
 * 
 * Usage:
 * - Build sets PRODUCT env variable (e.g., PRODUCT=gcxone)
 * - getProductConfig() returns current product config
 * - PostHog group() calls use product.groupKey for tagging
 */

export interface ProductConfig {
  id: string;           // Environment variable value (e.g., 'gcxone')
  name: string;         // Display name (e.g., 'GCXONE')
  groupKey: string;     // PostHog group identifier (e.g., 'gcxone')
  tagline?: string;     // Product tagline
  domain?: string;      // Product domain
}

/**
 * Product registry - add new products here
 */
export const PRODUCTS: Record<string, ProductConfig> = {
  gcxone: {
    id: 'gcxone',
    name: 'GCXONE',
    groupKey: 'gcxone',
    tagline: 'Complete documentation for NXGEN GCXONE platform',
    domain: 'docs.gcxone.com',
  },
  gcsurge: {
    id: 'gcsurge',
    name: 'GC Surge',
    groupKey: 'gcsurge',
    tagline: 'GC Surge Documentation',
    domain: 'docs.gcsurge.com',
  },
};

export const DEFAULT_PRODUCT: ProductConfig = {
  id: 'default',
  name: 'Default',
  groupKey: 'default',
};

/**
 * Get current product configuration from environment
 */
export function getProductConfig(): ProductConfig {
  const productId = typeof process !== 'undefined' 
    ? process.env.PRODUCT 
    : typeof import.meta !== 'undefined' 
      ? (import.meta as any).env?.PRODUCT 
      : undefined;
  
  if (productId && PRODUCTS[productId]) {
    return PRODUCTS[productId];
  }
  
  // Fallback to gcxone for backwards compatibility
  return PRODUCTS.gcxone;
}

/**
 * Get product config by ID
 */
export function getProductById(id: string): ProductConfig | undefined {
  return PRODUCTS[id];
}

/**
 * Get all registered products
 */
export function getAllProducts(): ProductConfig[] {
  return Object.values(PRODUCTS);
}
```

This approach ensures MPROD-05 requirement: new products can be added in under 1 day by simply extending the PRODUCTS registry.

```typescript
// Add to imports
import { getProductConfig } from '../utils/productConfig';

// Update AnalyticsWrapper component:
function AnalyticsWrapper({ children }: { children: React.ReactNode }) {
  const posthog = usePostHog();
  const productConfig = getProductConfig();

  useEffect(() => {
    if (posthog && typeof window !== 'undefined') {
      // Tag all events with product context
      posthog.group('product', productConfig.groupKey, {
        name: productConfig.name,
        id: productConfig.id,
      });
    }
  }, [posthog, productConfig]);

  useEffect(() => {
    if (posthog && typeof window !== 'undefined') {
      posthog.capture('$pageview', {
        product: productConfig.id,
      });
    }
  }, [posthog, productConfig]);

  return <>{children}</>;
}
```

This ensures ALL events captured after the group() call are automatically tagged with the product context in PostHog.
