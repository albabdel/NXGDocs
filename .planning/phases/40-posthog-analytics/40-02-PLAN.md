// Add import at top
import { getProductConfig } from '../utils/productConfig';

// Add new functions after existing exports:

/**
 * Track documentation page view with product context
 */
export function useTrackDocView() {
  const { capture } = usePostHog();
  
  return useCallback((docId: string, docTitle: string, docPath: string, category?: string) => {
    const productConfig = getProductConfig();
    capture(EVENT_NAMES.DOC_VIEW, {
      doc_id: docId,
      doc_title: docTitle,
      doc_path: docPath,
      category,
      product: productConfig.id,
    });
  }, [capture]);
}

/**
 * Track search query with product context
 */
export function useTrackSearch() {
  const { capture } = usePostHog();
  
  return useCallback((query: string, resultCount: number, searchMode: string = 'keyword') => {
    const productConfig = getProductConfig();
    capture(EVENT_NAMES.SEARCH_QUERY, {
      query: query.toLowerCase(),
      result_count: resultCount,
      search_mode: searchMode,
      product: productConfig.id,
    });
  }, [capture]);
}

/**
 * Track search result click with position and product
 */
export function useTrackSearchClick() {
  const { capture } = usePostHog();
  
  return useCallback((
    query: string,
    resultId: string,
    resultTitle: string,
    resultUrl: string,
    position: number
  ) => {
    const productConfig = getProductConfig();
    capture(EVENT_NAMES.SEARCH_RESULT_CLICK, {
      query: query.toLowerCase(),
      result_id: resultId,
      result_title: resultTitle,
      result_url: resultUrl,
      position,
      product: productConfig.id,
    });
  }, [capture]);
}
```

These hooks automatically include product context from the configuration.

// Add to imports
import { useTrackSearch, useTrackSearchClick } from '../../hooks/usePostHog';

// Inside component, add hooks:
export default function SearchModal() {
  // ... existing state
  const trackSearch = useTrackSearch();
  const trackSearchClick = useTrackSearchClick();
  
  // ... existing code
  
  // In the useEffect that handles debouncedQuery, add PostHog tracking:
  useEffect(() => {
    if (!fuse || !debouncedQuery.trim()) {
      setResults([]);
      setSelectedIndex(0);
      return;
    }

    const expandedTerms = expandQuery(debouncedQuery);
    const searchPattern = expandedTerms.join(' ');
    const rawResults = fuse.search(searchPattern, { limit: 50 });
    
    setResults(rawResults);
    setSelectedIndex(0);

    // Track search to PostHog with product context
    trackSearch(debouncedQuery, rawResults.length, useHybridSearchMode ? 'hybrid' : 'keyword');
    
    // Existing local analytics tracking
    trackSearchEnhanced(debouncedQuery, rawResults.length, useHybridSearchMode ? 'hybrid' : 'keyword', showAIPanel && isSemanticEnabled && rawResults.length > 0);
    
    if (rawResults.length > 0) {
      addRecentSearch(debouncedQuery);
    }
  }, [debouncedQuery, fuse, trackSearch, trackSearchEnhanced, addRecentSearch, useHybridSearchMode, showAIPanel]);
  
  // In handleResultClick, add PostHog tracking:
  const handleResultClick = useCallback((url: string, item?: SearchRecord, position?: number) => {
    if (debouncedQuery.trim()) {
      addRecentSearch(debouncedQuery);
    }
    
    // Track click to PostHog with product context
    if (item && position !== undefined) {
      trackSearchClick(debouncedQuery, item.id, item.title, url, position);
    }
    
    const searchId = getCurrentSearchId();
    if (searchId && item && position !== undefined) {
      trackClick(searchId, item.id, item.title, url, position);
    }
    close();
  }, [debouncedQuery, addRecentSearch, close, getCurrentSearchId, trackClick, trackSearchClick]);
}
```

This ensures search analytics flow to both local storage (for UI features like recent searches) and PostHog (for product-scoped insights).
