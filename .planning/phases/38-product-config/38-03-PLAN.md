   ```typescript
   import React, { createContext, useContext } from 'react';
   // @ts-ignore
   import { getCurrentConfig, getCurrentProduct, ProductConfig } from '../../product.config';
   
   interface ProductContextValue {
     productId: string;
     config: ProductConfig;
   }
   
   const ProductContext = createContext<ProductContextValue | null>(null);
   
   export function useProduct() {
     const context = useContext(ProductContext);
     if (!context) {
       throw new Error('useProduct must be used within ProductProvider');
     }
     return context;
   }
   ```

2. Wrap children in ProductContext.Provider:

   ```typescript
   const productConfig = getCurrentConfig();

         </AdminAuthProvider>
       </ProductContext.Provider>
     </PostHogProvider>
   );
   ```

3. Export the useProduct hook for other components to use.
