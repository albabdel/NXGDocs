Create Cloudflare Pages Function to handle Zoho OAuth callback for admin authentication.

Reference existing OAuth flow in `classic/src/components/ZohoTickets/useZohoAuth.ts`.

Flow:
1. Receive OAuth callback with `code` and `state` parameters
2. Exchange code for access token via Zoho token endpoint
3. Fetch user profile from Zoho (verify org membership)
4. Create admin session cookie
5. Redirect to `/admin` dashboard

Implementation:

```typescript
// functions/admin-auth-callback.ts
export async function onRequest(context: FunctionContext) {
  const { request, env } = context;
  const url = new URL(request.url);
  
  const code = url.searchParams.get('code');
  const state = url.searchParams.get('state');
  
  if (!code) {
    return new Response('Missing authorization code', { status: 400 });
  }
  
  // Exchange code for token
  const tokenResponse = await fetch('https://accounts.zoho.eu/oauth/v2/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'authorization_code',
      client_id: env.ZOHO_CLIENT_ID,
      client_secret: env.ZOHO_CLIENT_SECRET,
      redirect_uri: `${url.origin}/admin-auth-callback`,
      code,
    }),
  });
  
  const tokens = await tokenResponse.json();
  
  // Fetch user profile
  const profileResponse = await fetch('https://accounts.zoho.eu/oauth/v2/userinfo', {
    headers: { Authorization: `Bearer ${tokens.access_token}` },
  });
  
  const profile = await profileResponse.json();
  
  // Verify org membership
  if (profile.OrganizationId !== env.ZOHO_ORG_ID) {
    return new Response('Unauthorized: Not a member of this organization', { status: 403 });
  }
  
  // Create session
  const sessionToken = await createAdminSession(request, env, {
    id: profile.id,
    email: profile.email,
    name: profile.name,
    orgId: profile.OrganizationId,
  });
  
  // Log authentication
  await logAuthEvent(env, {
    type: 'admin.login',
    userId: profile.id,
    email: profile.email,
    success: true,
  });
  
  // Redirect to admin dashboard with session cookie
  return new Response(null, {
    status: 302,
    headers: {
      'Location': '/admin',
      'Set-Cookie': sessionToken,
    },
  });
}
```

```typescript
// classic/src/contexts/AdminAuthContext.tsx
import React, { createContext, useContext, useState, useEffect, useCallback } from 'react';

interface AdminUser {
  id: string;
  email: string;
  name: string;
  orgId: string;
  role: 'admin';
}

interface AdminAuthContextValue {
  user: AdminUser | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  login: () => void;
  logout: () => Promise<void>;
  refreshSession: () => Promise<void>;
}

const AdminAuthContext = createContext<AdminAuthContextValue | null>(null);

export function AdminAuthProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = useState<AdminUser | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  // Check session on mount
  useEffect(() => {
    validateSession();
  }, []);

  const validateSession = async () => {
    try {
      const response = await fetch('/admin-session', {
        credentials: 'include', // Include HttpOnly cookie
      });
      
      if (response.ok) {
        const data = await response.json();
        setUser(data.user);
      } else {
        setUser(null);
      }
    } catch (error) {
      console.error('Session validation failed:', error);
      setUser(null);
    } finally {
      setIsLoading(false);
    }
  };

  const login = useCallback(() => {
    const redirectUri = `${window.location.origin}/admin-auth-callback`;
    window.location.href = getAdminLoginUrl(redirectUri);
  }, []);

  const logout = useCallback(async () => {
    await fetch('/admin-logout', { method: 'POST', credentials: 'include' });
    setUser(null);
    window.location.href = '/';
  }, []);

  const refreshSession = useCallback(async () => {
    await validateSession();
  }, []);

  return (
    <AdminAuthContext.Provider value={{
      user,
      isAuthenticated: !!user,
      isLoading,
      login,
      logout,
      refreshSession,
    }}>
      {children}
    </AdminAuthContext.Provider>
  );
}

export function useAdminAuth() {
  const context = useContext(AdminAuthContext);
  if (!context) {
    throw new Error('useAdminAuth must be used within AdminAuthProvider');
  }
  return context;
}
```

```typescript
interface ProtectedRouteProps {
  children: React.ReactNode;

export function ProtectedRoute({ children }: ProtectedRouteProps) {
  const { isAuthenticated, isLoading } = useAdminAuth();

  // Show loading state while checking session
  if (isLoading) {
    return (
      <div className="flex items-center justify-center min-h-screen">
        <Loader className="w-8 h-8 animate-spin" style={{ color: '#E8B058' }} />
      </div>
    );
  }

  // Redirect to login if not authenticated
  if (!isAuthenticated) {
    return <Redirect to="/admin-login" />;
  }

  // Render protected content
  return <>{children}</>;
}
```

```typescript
// functions/lib/sync-admin-user.ts
export async function syncAdminUser(env: Env, user: AdminUser): Promise<void> {
  const client = getSanityClient(env);
  
  // Check if user exists
  const existing = await client.fetch(
    `*[_type == "adminUser" && zohoId == $zohoId][0]`,
    { zohoId: user.id }
  );
  
  if (existing) {
    // Update last login
    await client.patch(existing._id)
      .set({
        lastLoginAt: new Date().toISOString(),
        loginCount: (existing.loginCount || 0) + 1,
      })
      .commit();
  } else {
    // Create new user
    await client.create({
      _type: 'adminUser',
      zohoId: user.id,
      email: user.email,
      name: user.name,
      role: 'admin', // All org members are admins initially
      orgId: user.orgId,
      lastLoginAt: new Date().toISOString(),
      loginCount: 1,
      createdAt: new Date().toISOString(),
      isActive: true,
    });
  }
}
```
